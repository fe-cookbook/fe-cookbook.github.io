实现一个 bind 函数

const module = {
  x: 42,
  getX: function () {
    return this.x;
  },
};
const fn = module.getX;
console.info(fn());
fn.bind(module)();

简易版本

function bind(fn, context) {
  return function () {
    return fn.apply(context, arguments);
  };
}

升级版本

function bind(fn, context) {
  const outterArgs = Array.prototype.slice.call(arguments, 2);
  return function () {
    const innerArgs = [...arguments];
    return fn.apply(context, outterArgs.concat(innerArgs));
  };
}

const module = {
  x: 42,
  getX: function () {
    return this.x + JSON.stringify(arguments);
  },
};
const fn = module.getX;
bind(fn, module, "outter1", "outter2")("innter1", "inner2");
// "42 {"0":"outter1","1":"outter2","2":"innter1","3":"inner2"}"
